5.1 Cálculo de R² para el modelo

type <- rep("g", 5)
level <- rep(1, 5)
error_Model <- mgm_error_metrics(data = df_variables,
                                 type = type,
                                 level = level)
##   |                                                                              |                                                                      |   0%  |                                                                              |--------------                                                        |  20%  |                                                                              |----------------------------                                          |  40%  |                                                                              |------------------------------------------                            |  60%  |                                                                              |--------------------------------------------------------              |  80%  |                                                                              |----------------------------------------------------------------------| 100%
